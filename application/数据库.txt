#用户基本表
CREATE TABLE `graduation_project`.`user_base_info` ( `id` INT NOT NULL AUTO_INCREMENT COMMENT '用户id' , `name` VARCHAR(50) CHARACTER SET utf8 COLLATE utf8_general_ci NULL COMMENT '用户名' , `mobile` VARCHAR(11) CHARACTER SET utf8 COLLATE utf8_general_ci NULL COMMENT '手机号' , `accout` VARCHAR(255) CHARACTER SET utf8 COLLATE utf8_general_ci NULL COMMENT '用户账号' , `password` VARCHAR(255) CHARACTER SET utf8 COLLATE utf8_general_ci NULL COMMENT '用户密码' , `salt` VARCHAR(255) CHARACTER SET utf8 COLLATE utf8_general_ci NULL COMMENT '盐值' , `status` TINYINT(1) NULL DEFAULT '1' COMMENT '用户状态 -1注销，0冻结，1正常，' , `ip` VARCHAR(20) CHARACTER SET utf8 COLLATE utf8_general_ci NULL COMMENT '用户ip' , `last_login_time` TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP COMMENT '用户最后登录时间' , `creation_time` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '用户创建时间' , `modification_time` DATETIME on update CURRENT_TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , PRIMARY KEY (`id`)) ENGINE = InnoDB CHARSET=utf8 COLLATE utf8_general_ci COMMENT = '用户基本信息表';

#用户个人信息表
CREATE TABLE `graduation_project`.`user_person_info` ( `id` INT NOT NULL AUTO_INCREMENT , `user_id` INT NOT NULL COMMENT '用户id' , `nickname` VARCHAR(255) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL COMMENT '用户昵称' , `description` VARCHAR(255) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL COMMENT '用户简介' , `image` VARCHAR(255) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL COMMENT '用户头像地址' , `creation_time` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , `modification_time` TIMESTAMP on update CURRENT_TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , PRIMARY KEY (`id`)) ENGINE = InnoDB CHARSET=utf8 COLLATE utf8_general_ci COMMENT = '用户个人信息表';

#用户文章表0-9
CREATE TABLE `graduation_project`.`user_article_0` ( `id` INT NOT NULL AUTO_INCREMENT , `user_id` INT NOT NULL COMMENT '用户id' , `author` VARCHAR(255) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL COMMENT '用户昵称' , `name` VARCHAR(255) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL COMMENT '文章名' , `status` TINYINT(1) NOT NULL COMMENT '文章状态' , `content` TEXT CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL COMMENT '文章内容' , `creation_time` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , `modification_time` TIMESTAMP on update CURRENT_TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , PRIMARY KEY (`id`)) ENGINE = InnoDB CHARSET=utf8 COLLATE utf8_general_ci COMMENT = '用户文章表';
